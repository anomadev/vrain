## Ejemplos PrÃĄcticos

```php
namespace App\Livewire;

use Livewire\Component;
use App\Models\Post;

class CreatePost extends Component
{
    public string $title = ''; // Public property = Reactive state

    public function save() // Action
    {
        $this->validate(['title' => 'required|min:3']);

    }
```

```html
    @endif

```

## Buenas PrÃĄcticas
